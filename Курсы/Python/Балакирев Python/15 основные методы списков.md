```python
lst.append(100)#добавит элемент 100 в конец списка

lst.insert(3, 100) #вставляет 100 вместо элемента с индесом 3

lst.pop(3)#без аргументов удаляет последний элемент, как аргумент принимает индекс(и удаляет нужный элемент)

lst.clear()#очищает весь список

lst.count(100)#находит кол-во элементов 100 

lst.remove(100)#удаляет все элементы 100

lst.index(100, 3)#определяет индекс элемента 100, ищет с индекса 3 

lst.reverse()#разворачивает список( как lst[::-1])

lst.sort()#сортирует список по возрастанию(меняет список, а не возвращает новый)

lst2 = lst.copy()#копирует список(создается новый)(оно же lst2 = lst[:])
```
