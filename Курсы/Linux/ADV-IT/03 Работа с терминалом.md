
```bash
uptime   - время с последнего запуска

uname -a - узнать какой линукс стоит

lscpu - информация о процессоре

clear - очистить терминал

ls -l(ls) - вывести содержимое папки

man – помощь

info – тоже помощь

whatis – показывает что делает комманда

whereis – показывает где файл

echo 'Hello World' - выводит текст(без ковычек тоже работает)

Ctrl + c - прерывает работу

ps - показывает активные процессы

Ctrl + z - убирает процесс на задний план

fg - возвращает процесс на передний план

ls –la –R / - показать все на компьютере

tree - показывает файловое дерево текущей директории(или можно указать путь)

```
